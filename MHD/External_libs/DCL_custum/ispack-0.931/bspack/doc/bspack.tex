\section{概要}

これは他のサブルーチンパッケージで使用される基礎的なサブルーチンを集め
たパッケージである.

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\section{サブルーチンのリスト}

\vspace{1em}
\begin{tabular}{ll}
{\tt BSDMSG} & メッセージの出力\\
{\tt BSSET0} & 配列に0を代入\\
{\tt BSCOPY} & 配列の複写
\end{tabular}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\section{サブルーチンの説明}

\subsection{BSDMSG}

\begin{enumerate}

\item 機能 

メッセージの出力

\item 定義

\item 呼び出し方法 
    
{\tt BSDMSG(CL,CS,CM)}
  
\item パラメーターの説明 
    
\begin{tabular}{lll}
{\tt CL} & {\tt (C*1)} & 入力. メッセージの種類の指定(備考を参照)\\
{\tt CS} & {\tt (C*(*))} & 入力. サブルーチン名\\
{\tt CM} & {\tt (C*(*))} & 入力. メッセージの内容
を参照)
\end{tabular}

\item 備考

(a) {\tt CL}が{\tt 'E'}ならエラーが生じたと見なして処理を終了,
    {\tt 'W'}ならワーニングを出力して処理を続行, 
    {\tt 'M'}なら単にメッセージを出力する.

\end{enumerate}

%---------------------------------------------------------------------

\subsection{BSSET0}

\begin{enumerate}

\item 機能 

配列に0を代入

\item 定義

\item 呼び出し方法 
    
{\tt BSSET0(N,A)}
  
\item パラメーターの説明 
    
\begin{tabular}{lll}
{\tt N} & {\tt (I)} & 入力. 配列{\tt A}の大きさ\\
{\tt A} & {\tt (D(N))} & 出力. すべての要素に0が代入される\\
\end{tabular}

\item 備考

\end{enumerate}

%---------------------------------------------------------------------

\subsection{BSCOPY}

\begin{enumerate}

\item 機能 

配列の複写

\item 定義

\item 呼び出し方法 
    
{\tt BSCOPY(N,A,B)}
  
\item パラメーターの説明 
    
\begin{tabular}{lll}
{\tt N} & {\tt (I)} & 入力. 配列{\tt A, B}の大きさ\\
{\tt A} & {\tt (D(N))} & 入力\\
{\tt B} & {\tt (D(N))} & 出力. 配列{\tt A}の内容がそのまま複写される.
\end{tabular}

\item 備考

\end{enumerate}
