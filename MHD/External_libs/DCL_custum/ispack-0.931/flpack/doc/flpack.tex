\section{概要}

文字コードおよび実数表現の変換を行う.

\section{サブルーチンのリスト}
\begin{tabular}{lp{10cm}}
  {\tt FLCECA(C1,C2)}  & 文字コードの変換 (EBCDIC $\to$ ASCII) \\
  {\tt FLCACE(C1,C2)}  & 文字コードの変換 (ASCII $\to$ EBCDIC) \\
  {\tt FLCKCA(C1,C2)}  & 文字コードの変換 (EBCDIK $\to$ ASCII) \\
  {\tt FLCACK(C1,C2)}  & 文字コードの変換 (ASCII $\to$ EBCDIK) \\
  {\tt FLRBRE(I1,I2)}  & 実数表現の変換 (IBM(REAL*4) $\to$ IEEE(REAL*4)) \\
  {\tt FLRERB(I1,I2)}  & 実数表現の変換 (IEEE(REAL*4) $\to$ IBM(REAL*4)) \\
  {\tt FLDBDE(I1,I2)}  & 実数表現の変換 (IBM(REAL*8) $\to$ IEEE(REAL*8)) \\
  {\tt FLDEDB(I1,I2)}  & 実数表現の変換 (IEEE(REAL*8) $\to$ IBM(REAL*8)) \\
  {\tt FLDBRE(I1,I2)}  & 実数表現の変換 (IBM(REAL*8) $\to$ IEEE(REAL*4)) \\
  {\tt FLREDB(I1,I2)}  & 実数表現の変換 (IEEE(REAL*4) $\to$ IBM(REAL*8)) \\
  {\tt FLRBDE(I1,I2)}  & 実数表現の変換 (IBM(REAL*4) $\to$ IEEE(REAL*8)) \\
  {\tt FLDERB(I1,I2)}  & 実数表現の変換 (IEEE(REAL*8) $\to$ IBM(REAL*4))
\end{tabular}

\section{サブルーチンの説明}

\subsection{FLCECA/FLCACE/FLCKCA/FLCACK}
\begin{enumerate}
  \item 機能
  \begin{quote}
    文字コードを変換する.\\
    \begin{tabular}{lp{10cm}}
     {\tt FLCECA(C1,C2)}  & 文字コードの変換 (EBCDIC $\to$ ASCII) \\
     {\tt FLCACE(C1,C2)}  & 文字コードの変換 (ASCII $\to$ EBCDIC) \\
     {\tt FLCKCA(C1,C2)}  & 文字コードの変換 (EBCDIK $\to$ ASCII) \\
     {\tt FLCACK(C1,C2)}  & 文字コードの変換 (ASCII $\to$ EBCDIK)
  \end{tabular}
  \end{quote}
  \item 呼び出し方法
  \begin{quote}
    {\tt CALL FLCECA(C1,C2)}\\
    {\tt CALL FLCACE(C1,C2)}\\
    {\tt CALL FLCKCA(C1,C2)}\\
    {\tt CALL FLCACK(C1,C2)}
  \end{quote}
  \item パラメーターの説明
  \begin{quote}
    \begin{tabular}{llp{10cm}}
      {\tt C1} & {\tt (C*1)} & 変換する文字コードを入力する文字型の引数\\
      {\tt C2} & {\tt (C*1)} & 変換した文字コードを返す文字型の引数
    \end{tabular}
  \end{quote}
  \item 備考
  \begin{enumerate}
    \item ASCIIコードで定義されていない番号(128番目以降)については
          適当な番号づけをすることによって, ASCII $\leftrightarrow$
          EBCDIC(K)の対応を1対1にして, 逆変換での再現性を保証している.
  \end{enumerate}
\end{enumerate}

\subsection{FLRBRE/FLRERB/FLDBDE/FLDEDB/FLDBRE/FLREDB/FLRBDE/FLDERB}
\begin{enumerate}
  \item 機能
  \begin{quote}
    実数表現を変換する.\\
    \begin{tabular}{lp{10cm}}
     {\tt FLRBRE(I1,I2)}  & 実数表現の変換 (IBM(REAL*4) $\to$ IEEE(REAL*4)) \\
     {\tt FLRERB(I1,I2)}  & 実数表現の変換 (IEEE(REAL*4) $\to$ IBM(REAL*4)) \\
     {\tt FLDBDE(I1,I2)}  & 実数表現の変換 (IBM(REAL*8) $\to$ IEEE(REAL*8)) \\
     {\tt FLDEDB(I1,I2)}  & 実数表現の変換 (IEEE(REAL*8) $\to$ IBM(REAL*8)) \\
     {\tt FLDBRE(I1,I2)}  & 実数表現の変換 (IBM(REAL*8) $\to$ IEEE(REAL*4)) \\
     {\tt FLREDB(I1,I2)}  & 実数表現の変換 (IEEE(REAL*4) $\to$ IBM(REAL*8)) \\
     {\tt FLRBDE(I1,I2)}  & 実数表現の変換 (IBM(REAL*4) $\to$ IEEE(REAL*8)) \\
     {\tt FLDERB(I1,I2)}  & 実数表現の変換 (IEEE(REAL*8) $\to$ IBM(REAL*4))
    \end{tabular}
  \end{quote}
  \item 呼び出し方法
  \begin{quote}
     {\tt CALL FLRBRE(I1,I2)} \\
     {\tt CALL FLRERB(I1,I2)} \\
     {\tt CALL FLDBDE(I1,I2)} \\
     {\tt CALL FLDEDB(I1,I2)} \\
     {\tt CALL FLDBRE(I1,I2)} \\
     {\tt CALL FLREDB(I1,I2)} \\
     {\tt CALL FLRBDE(I1,I2)} \\
     {\tt CALL FLDERB(I1,I2)}
  \end{quote}
  \item パラメーターの説明
  \begin{quote}
    \begin{tabular}{llp{10cm}}
      {\tt I1} & {\tt (I)} & 入力する実数表現を格納した整数型の引数. 
                             入力が倍精度(REAL*8)なら長さ2の配列とする.\\
      {\tt I2} & {\tt (I)} & 変換した実数表現を返す整数型の引数.
                            出力が倍精度(REAL*8)なら長さ2の配列とする.
    \end{tabular}
  \end{quote}
  \item 備考
  \begin{enumerate}
    \item 変換においては, 表現しうる最も近い数への変換が行われる. 
         すなわち, 有効ビット数が減少するような変換においては, 
          0捨1入の丸めを行う.
    \item IBM形式では無限大に対応する表現が存在しないので, IEEE形式の
          入力が無限大またはIBM形式で表現できる最大数を超えるような場
          合には, 変換結果をIBM形式で表現できる最大数とする. 
          逆に, IBM形式の入力がIEEE形式で表現できる最大数を超えるよう
          な場合には, 変換結果はIEEE形式の表現における無限大とする.
    \item 非数表現が入力された場合には, その旨のメッセージを出力して
          強制終了する.
  \end{enumerate}
\end{enumerate}
